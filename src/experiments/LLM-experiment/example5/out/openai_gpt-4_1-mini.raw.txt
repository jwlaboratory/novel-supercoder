	.arch armv8-a
	.file	"inputC.c"
	.text
	.section	.rodata.str1.8,"aMS",@progbits,1
	.align	3
.LC0:
	.string	"%d %d"
	.align	3
.LC1:
	.string	"%d"
	.align	3
.LC2:
	.string	"%d "
	.align	3
.LC3:
	.string	"%d\n"
	.section	.text.startup,"ax",@progbits
	.align	2
	.p2align 4,,11
	.global	main
	.type	main, %function
main:
.LFB0:
	.cfi_startproc
	stp	x29, x30, [sp, -512]!
	.cfi_def_cfa_offset 512
	.cfi_offset 29, -512
	.cfi_offset 30, -504
	mov	x29, sp
	stp	x19, x20, [sp, 16]
	stp	x21, x22, [sp, 32]
	stp	x23, x24, [sp, 48]
	stp	x25, x26, [sp, 64]
	stp	x27, x28, [sp, 80]
	stp	x29, x30, [sp, 96]	// save again for aligned local area
	sub	sp, sp, 384		// allocate local arrays and buffers

	// local variables layout on stack:
	// [sp + 0]  : n[100] (400 bytes)
	// [sp + 400]: x buffer start (dynamic)
	// We'll allocate max 400 bytes for n, and dynamic for x after

	// zero initialize n[100]
	mov	x0, sp
	mov	x1, 0
	mov	x2, 400
	bl	memset

	// scanf("%d %d", &r, &c)
	adrp	x0, .LC0
	add	x0, x0, :lo12:.LC0
	sub	sp, sp, 32		// shadow space for scanf
	add	x1, sp, 16		// &r
	add	x2, sp, 20		// &c
	bl	__isoc99_scanf
	ldrsw	x19, [sp, 16]	// r
	ldrsw	x20, [sp, 20]	// c
	add	sp, sp, 32

	// if r <= 0 or c <= 0, skip input and output zeros
	cmp	x19, 1
	b.lt	.Lzero_output
	cmp	x20, 1
	b.lt	.Lzero_output

	// calculate size for x array: r*c*4 bytes, align 16
	mul	x21, x19, x20		// r*c
	lsl	x22, x21, 2		// bytes needed
	add	x22, x22, 15
	and	x22, x22, -16		// align 16

	// allocate x array on stack
	sub	sp, sp, x22

	// read x array
	mov	x23, 0			// i = 0
	mov	x24, sp			// x array base pointer
	mov	x25, x20		// c

.Lread_rows:
	cmp	x23, x19
	b.ge	.Lread_done
	mov	x26, 0			// j = 0

.Lread_cols:
	cmp	x26, x25
	b.ge	.Lnext_row
	adrp	x0, .LC1
	add	x0, x0, :lo12:.LC1
	add	x1, x24, x26, lsl 2
	bl	__isoc99_scanf
	add	x26, x26, 1
	b	.Lread_cols

.Lnext_row:
	add	x24, x24, x25, lsl 2
	add	x23, x23, 1
	b	.Lread_rows

.Lread_done:
	// zero initialize n array again (just in case)
	mov	x0, sp
	mov	x1, 0
	mov	x2, 400
	bl	memset

	// print rows with sums and accumulate column sums in n
	mov	x23, 0			// i = 0
	mov	x24, sp			// n array base pointer
	mov	x25, sp			// reuse n base pointer for n
	add	x26, sp, 400		// x array base pointer
	mov	x27, x20		// c

.Lprint_rows:
	cmp	x23, x19
	b.ge	.Lprint_cols
	mov	x28, 0			// j = 0
	mov	w21, 0			// z = 0

.Lprint_cols_loop:
	cmp	x28, x27
	b.ge	.Lprint_row_sum
	ldr	w0, [x26, x23, lsl 2, x28, lsl 2]	// x[i][j]
	// print "%d "
	adrp	x0, .LC2
	add	x0, x0, :lo12:.LC2
	mov	w1, w0
	bl	printf

	add	w21, w21, w0		// z += x[i][j]

	// n[j] += x[i][j]
	ldr	w2, [x25, x28, lsl 2]
	add	w2, w2, w0
	str	w2, [x25, x28, lsl 2]

	add	x28, x28, 1
	b	.Lprint_cols_loop

.Lprint_row_sum:
	// print "%d\n", z
	adrp	x0, .LC3
	add	x0, x0, :lo12:.LC3
	mov	w1, w21
	bl	printf

	add	x23, x23, 1
	b	.Lprint_rows

.Lprint_cols:
	// print column sums and total sum
	mov	x23, 0			// j = 0
	mov	w21, 0			// w = 0

.Lprint_cols_loop2:
	cmp	x23, x20
	b.ge	.Lend
	ldr	w0, [x25, x23, lsl 2]
	adrp	x1, .LC2
	add	x1, x1, :lo12:.LC2
	mov	x0, x1
	mov	w1, w0
	bl	printf

	add	w21, w21, w0
	add	x23, x23, 1
	b	.Lprint_cols_loop2

.Lend:
	// print total sum w
	adrp	x0, .LC3
	add	x0, x0, :lo12:.LC3
	mov	w1, w21
	bl	printf

	// restore stack and return 0
	add	sp, sp, x22
	add	sp, sp, 384
	ldp	x27, x28, [sp, 80]
	ldp	x25, x26, [sp, 64]
	ldp	x23, x24, [sp, 48]
	ldp	x21, x22, [sp, 32]
	ldp	x19, x20, [sp, 16]
	ldp	x29, x30, [sp], 512
	mov	w0, 0
	ret

.Lzero_output:
	// print zeros for invalid input
	adrp	x0, .LC3
	add	x0, x0, :lo12:.LC3
	mov	w1, 0
	bl	printf
	mov	w0, 0
	ldp	x19, x20, [sp, 16]
	ldp	x21, x22, [sp, 32]
	ldp	x23, x24, [sp, 48]
	ldp	x25, x26, [sp, 64]
	ldp	x27, x28, [sp, 80]
	ldp	x29, x30, [sp], 512
	ret
	.cfi_endproc
.LFE0:
	.size	main, .-main
	.ident	"GCC: (GNU) 13.4.0"
	.section	.note.GNU-stack,"",@progbits
