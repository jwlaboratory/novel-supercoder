void dump(int n)
{
	int i;

	printf("%d", a[0]);

void insertion_sort(int n)
{
	int i, j, key;

	dump(n);
	for (i = 1; i < n; i++) {
		key = a[i];
		j = i-1;
		while (j >= 0 && a[j] > key) a[j+1] = a[j], j--;
		a[j+1] = key;
		dump(n);
	}
}

int main()

	for (i = 0; i < N; i++) a[i] = in();
	insertion_sort(N);
	return 0;
