void AVL_set_inside_merge(AVL_set **root, AVL_set **node){
    if(*node == NULL) return;
    if(*root == NULL){
        *root = *node;
        *node = NULL;
        return;
    }
    if((*root)->val < (*node)->val){
        AVL_set_inside_merge(&(*root)->child[1], node);
    }else{
        AVL_set_inside_merge(&(*root)->child[0], node);
    }
    AVL_set_inside_update(root, (*root)->val, 0);
}
void AVL_set_inside_split(AVL_set **root, AVL_set **node, ll val){
    if(*root == NULL){
        *node = NULL;
        return;
    }
    if((*root)->val < val){
        AVL_set_inside_split(&(*root)->child[1], node, val);
        AVL_set_inside_merge(&(*root), node);
    }else{
        AVL_set_inside_split(&(*root)->child[0], node, val);
        AVL_set_inside_merge(&(*root), node);
    }
    AVL_set_inside_update(root, (*root)->val, 0);
}
